13  
14  	public MethodInjectorTestClassLoader(final ClassVisitor pVisitor, final ClassWriter pWriter,
15  			final Class<?> pTestSerializableClass, final Bundle pBundle) {
16  		super(new InspectForInjectorMethodClassVisitor(pVisitor), pWriter, pTestSerializableClass, pBundle);
17  	}
18  
19  	@Override
20  	protected byte[] secondPass(final byte[] pClassData) {
21  		final ClassReader reader = new ClassReader(writer.toByteArray());
22  		writer = new ClassWriter(reader, 0);
23  		visitor = new MethodInjectionClassVisitor(writer, (InspectForInjectorMethodClassVisitor) visitor);
24  		reader.accept(visitor, 0);
25  		return writer.toByteArray();
26  	}
