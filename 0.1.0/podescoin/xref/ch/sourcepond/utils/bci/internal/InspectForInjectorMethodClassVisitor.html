1   package ch.sourcepond.utils.bci.internal;

3   import static ch.sourcepond.utils.bci.internal.Constants.CONSTRUCTOR_NAME;
4   import static ch.sourcepond.utils.bci.internal.Constants.INJECT_ANNOTATION_NAME;
5   import static java.lang.String.format;
6   import static org.objectweb.asm.Opcodes.ASM5;
7   import static org.objectweb.asm.Type.getArgumentTypes;
8   
9   import org.objectweb.asm.ClassVisitor;
10  import org.objectweb.asm.MethodVisitor;
11  import org.objectweb.asm.Type;
12  
13  final class InspectForInjectorMethodClassVisitor extends ClassVisitor {
14  	private static final String[][] EMPTY = new String[0][0];
15  	private String[][] namedComponents;
16  	private String injectorMethodName;
17  	private String injectorMethodDesc;
18  
19  	InspectForInjectorMethodClassVisitor(final ClassVisitor pWriter) {
20  		super(ASM5, pWriter);
21  	}
22  
23  	@Override
24  	public MethodVisitor visitMethod(final int access, final String name, final String desc, final String signature,
25  			final String[] exceptions) {
26  		if (!CONSTRUCTOR_NAME.equals(name)) {
27  			return new InjectorMethodVisitor(this, super.visitMethod(access, name, desc, signature, exceptions), name,
28  					desc);
29  		}
30  		return super.visitMethod(access, name, desc, signature, exceptions);
31  	}
32  
33  	void addNamedComponent(final String pComponentId, final int pParameterIndex) {
34  		namedComponents[pParameterIndex][0] = pComponentId;
35  	}
36  
37  	boolean isArgumentTypesInitialized() {
38  		return namedComponents != null;
39  	}
40  
41  	String[][] getComponents() {
42  		return namedComponents;
43  	}
44  
45  	void initArgumentTypes(final String pInjectorMethodName, final String pInjectorMethodDesc) {
46  		if (isArgumentTypesInitialized()) {
47  			throw new AmbiguousInjectorMethodsException(
48  					format("More than one method detected which is annotated with %s", INJECT_ANNOTATION_NAME));
49  		}
50  
51  		injectorMethodName = pInjectorMethodName;
52  		injectorMethodDesc = pInjectorMethodDesc;
53  
54  		final Type[] argumentTypes = getArgumentTypes(pInjectorMethodDesc);
55  		if (argumentTypes.length > 0) {
56  			namedComponents = new String[argumentTypes.length][2];
57  			for (int i = 0; i < argumentTypes.length; i++) {
58  				namedComponents[i][1] = argumentTypes[i].getClassName();
59  			}
60  		} else {
61  			namedComponents = EMPTY;
62  		}
63  	}
64  
65  	public String getInjectorMethodName() {
66  		return injectorMethodName;
67  	}
68  
69  	public String getInjectorMethodDesc() {
70  		return injectorMethodDesc;
71  	}
72  }
