3   import static ch.sourcepond.utils.bci.internal.Constants.INJECT_ANNOTATION_NAME;
4   import static ch.sourcepond.utils.bci.internal.Constants.NAMED_ANNOTATION_NAME;
5   import static org.objectweb.asm.Opcodes.ASM5;
6   import static org.objectweb.asm.Type.getType;
7   
8   import org.objectweb.asm.AnnotationVisitor;
9   import org.objectweb.asm.MethodVisitor;
10  
11  final class InjectorMethodVisitor extends MethodVisitor {
12  	private final InspectForInjectorMethodClassVisitor classVisitor;
13  	private final String injectorMethodName;
14  	private final String injectorMethodDesc;
15  
16  	public InjectorMethodVisitor(final InspectForInjectorMethodClassVisitor pClassVisitor, final MethodVisitor mv,
17  			final String pInjectorMethodName, final String pInjectorMethodDesc) {
18  		super(ASM5, mv);
19  		classVisitor = pClassVisitor;
20  		injectorMethodName = pInjectorMethodName;
21  		injectorMethodDesc = pInjectorMethodDesc;
22  	}
23  
24  	@Override
25  	public AnnotationVisitor visitAnnotation(final String desc, final boolean visible) {
26  		if (visible && INJECT_ANNOTATION_NAME.equals(getType(desc).getClassName())) {
27  			classVisitor.initArgumentTypes(injectorMethodName, injectorMethodDesc);
28  		}
29  		return super.visitAnnotation(desc, visible);
30  	}
31  
32  	@Override
33  	public AnnotationVisitor visitParameterAnnotation(final int parameter, final String desc, final boolean visible) {
34  		if (classVisitor.isArgumentTypesInitialized()) {
35  			if (NAMED_ANNOTATION_NAME.equals(getType(desc).getClassName())) {
36  				return new NamedAnnotationOnParameterVisitor(this,
37  						super.visitParameterAnnotation(parameter, desc, visible), parameter);
38  			}
39  		}
40  		return super.visitParameterAnnotation(parameter, desc, visible);
41  	}
42  
43  	void setComponentId(final String pComponentId, final int pParameterIndex) {
44  		classVisitor.addNamedComponent(pComponentId, pParameterIndex);
45  	}
46  }
