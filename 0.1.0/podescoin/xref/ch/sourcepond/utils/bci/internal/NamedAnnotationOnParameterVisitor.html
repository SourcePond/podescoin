3   import static org.objectweb.asm.Opcodes.ASM5;
4   
5   import org.objectweb.asm.AnnotationVisitor;
6   
7   public class NamedAnnotationOnParameterVisitor extends AnnotationVisitor {
8   	private final InjectorMethodVisitor methodVisitor;
9   	private final int parameterIndex;
10  
11  	NamedAnnotationOnParameterVisitor(final InjectorMethodVisitor pMethodVisitor,
12  			final AnnotationVisitor pDelegate, final int pParameterIndex) {
13  		super(ASM5, pDelegate);
14  		methodVisitor = pMethodVisitor;
15  		parameterIndex = pParameterIndex;
16  	}
17  
18  	@Override
19  	public void visit(final String name, final Object value) {
20  		methodVisitor.setComponentId((String) value, parameterIndex);
21  		super.visit(name, value);
22  	}
23  }
