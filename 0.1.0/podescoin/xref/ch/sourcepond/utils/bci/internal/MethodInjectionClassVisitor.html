24  import org.objectweb.asm.ClassVisitor;
25  import org.objectweb.asm.Label;
26  import org.objectweb.asm.MethodVisitor;
27  import org.objectweb.asm.Opcodes;

29  import ch.sourcepond.utils.bci.Container;

31  class MethodInjectionClassVisitor extends SerializableClassVisitor {
32  	private static final String CONTAINER_INTERNAL_NAME = getInternalName(Container.class);
33  	private static final String GET_CONTAINER_METHOD_NAME = "getContainer";
34  	private static final String GET_CONTAINER_METHOD_DESC = getMethodDescriptor(getType(Container.class),
35  			getType(Serializable.class));
36  	private static final String GET_COMPONENT_BY_ID_NAME = "getComponentById";
37  	private static final String GET_COMPONENT_BY_ID_DESC = getMethodDescriptor(getType(Object.class),
38  			getType(String.class), getType(String.class), getType(int.class));
39  	private static final String GET_COMPONENT_BY_TYPE_NAME_NAME = "getComponentByTypeName";
40  	private static final String GET_COMPONENT_BY_TYPE_DESC = getMethodDescriptor(getType(Object.class),
41  			getType(String.class), getType(int.class));
42  	private static final String OBJECT_INPUT_STREAM_INTERNAL_NAME = getInternalName(ObjectInputStream.class);
43  	private static final String EXCEPTION_INTERNAL_NAME = getInternalName(Exception.class);
44  	private static final String ILLEGAL_EXCEPTION_INTERNAL_NAME = getInternalName(IllegalStateException.class);
45  	private static final String GET_MESSAGE_NAME = "getMessage";
46  	private static final String GET_MESSAGE_DESC = getMethodDescriptor(getType(String.class));
47  	private static final String CONSTRUCTOR_DESC = getMethodDescriptor(getType(void.class), getType(String.class),
48  			getType(Throwable.class));
49  	private static final int MIN_STACK_SIZE = 4;
50  	private final InspectForInjectorMethodClassVisitor inspector;
51  
52  	public MethodInjectionClassVisitor(final ClassVisitor pVisitor,
53  			final InspectForInjectorMethodClassVisitor pInspector) {
54  		super(pVisitor);
55  		inspector = pInspector;
56  	}
57  
58  	@Override
59  	protected boolean isEnhancementNecessary() {
60  		final String[][] components = inspector.getComponents();
61  		return components != null && components.length > 0;
62  	}
63  
64  	@Override
65  	protected void enhanceReadObject(final MethodVisitor mv) {
66  		mv.visitCode();
67  		final Label l0 = new Label();
68  		final Label l1 = new Label();
69  		final Label l2 = new Label();
70  		mv.visitTryCatchBlock(l0, l1, l2, EXCEPTION_INTERNAL_NAME);
71  		mv.visitVarInsn(ALOAD, 0);
72  		mv.visitMethodInsn(INVOKESTATIC, INJECTOR_INTERNAL_NAME, GET_CONTAINER_METHOD_NAME, GET_CONTAINER_METHOD_DESC,
73  				false);
74  		mv.visitVarInsn(ASTORE, 2);
75  		mv.visitLabel(l0);
76  		mv.visitVarInsn(ALOAD, 0);
77  
78  		final String[][] components = inspector.getComponents();
79  
80  		boolean increaseByOne = false;
81  		int stackSize = MIN_STACK_SIZE;
82  		for (int i = 0; i < components.length; i++, stackSize++) {
83  			mv.visitVarInsn(ALOAD, 2);
84  			if (components[i][0] != null) {
85  				mv.visitLdcInsn(components[i][0]);
86  				mv.visitLdcInsn(components[i][1]);
87  				pushByteConstant(mv, i);
88  				mv.visitMethodInsn(INVOKEINTERFACE, CONTAINER_INTERNAL_NAME, GET_COMPONENT_BY_ID_NAME,
89  						GET_COMPONENT_BY_ID_DESC, true);
90  
91  				if (!increaseByOne) {
92  					increaseByOne = true;
93  				}
94  			} else {
95  				mv.visitLdcInsn(components[i][1]);
96  				pushByteConstant(mv, i);
97  				mv.visitMethodInsn(INVOKEINTERFACE, CONTAINER_INTERNAL_NAME, GET_COMPONENT_BY_TYPE_NAME_NAME,
98  						GET_COMPONENT_BY_TYPE_DESC, true);
99  			}
100 			mv.visitTypeInsn(CHECKCAST, components[i][1].replace('.', '/'));
101 		}
102 
103 		mv.visitMethodInsn(INVOKEVIRTUAL, thisClassInternalName, inspector.getInjectorMethodName(),
104 				inspector.getInjectorMethodDesc(), false);
105 
106 		mv.visitLabel(l1);
107 		final Label l3 = new Label();
108 		mv.visitJumpInsn(GOTO, l3);
109 		mv.visitLabel(l2);
110 		mv.visitFrame(Opcodes.F_FULL, 3,
111 				new Object[] { thisClassInternalName, OBJECT_INPUT_STREAM_INTERNAL_NAME, CONTAINER_INTERNAL_NAME }, 1,
112 				new Object[] { EXCEPTION_INTERNAL_NAME });
113 		mv.visitVarInsn(ASTORE, 3);
114 		mv.visitTypeInsn(NEW, ILLEGAL_EXCEPTION_INTERNAL_NAME);
115 		mv.visitInsn(DUP);
116 		mv.visitVarInsn(ALOAD, 3);
117 		mv.visitMethodInsn(INVOKEVIRTUAL, EXCEPTION_INTERNAL_NAME, GET_MESSAGE_NAME, GET_MESSAGE_DESC, false);
118 		mv.visitVarInsn(ALOAD, 3);
119 		mv.visitMethodInsn(INVOKESPECIAL, ILLEGAL_EXCEPTION_INTERNAL_NAME, CONSTRUCTOR_NAME, CONSTRUCTOR_DESC, false);
120 		mv.visitInsn(ATHROW);
121 		mv.visitLabel(l3);
122 		mv.visitFrame(Opcodes.F_SAME, 0, null, 0, null);
123 		mv.visitInsn(RETURN);
124 		mv.visitMaxs(increaseByOne ? stackSize + 1 : stackSize, 4);
125 		mv.visitEnd();
126 	}
127 }
