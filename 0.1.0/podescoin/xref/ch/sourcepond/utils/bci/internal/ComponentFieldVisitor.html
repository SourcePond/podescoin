1   package ch.sourcepond.utils.bci.internal;

3   import static ch.sourcepond.utils.bci.internal.Constants.INJECT_ANNOTATION_NAME;
4   import static ch.sourcepond.utils.bci.internal.Constants.NAMED_ANNOTATION_NAME;
5   import static org.objectweb.asm.Opcodes.ASM5;
6   import static org.objectweb.asm.Type.getType;
7   
8   import org.objectweb.asm.AnnotationVisitor;
9   import org.objectweb.asm.FieldVisitor;
10  
11  final class ComponentFieldVisitor extends FieldVisitor {
12  	private final FieldInjectionClassVisitor classVisitor;
13  	private final String fieldName;
14  	private final String fieldType;
15  	private String componentIdOrNull;
16  	private boolean inject;
17  
18  	ComponentFieldVisitor(final FieldInjectionClassVisitor pClassVisitor, final FieldVisitor pDelegate,
19  			final String pFieldName, final String pFieldType) {
20  		super(ASM5, pDelegate);
21  		classVisitor = pClassVisitor;
22  		fieldName = pFieldName;
23  		fieldType = pFieldType;
24  	}
25  
26  	void setComponentId(final String pComponentId) {
27  		componentIdOrNull = pComponentId;
28  	}
29  
30  	@Override
31  	public AnnotationVisitor visitAnnotation(final String desc, final boolean visible) {
32  		if (visible) {
33  			if (INJECT_ANNOTATION_NAME.equals(getType(desc).getClassName())) {
34  				inject = true;
35  			}
36  			if (NAMED_ANNOTATION_NAME.equals(getType(desc).getClassName())) {
37  				return new NamedAnnotationOnFieldVisitor(this, fv.visitAnnotation(desc, visible));
38  			}
39  		}
40  		return super.visitAnnotation(desc, visible);
41  	}
42  
43  	@Override
44  	public void visitEnd() {
45  		if (inject) {
46  			classVisitor.addNamedComponent(fieldName, componentIdOrNull, fieldType);
47  		}
48  	}
49  }
