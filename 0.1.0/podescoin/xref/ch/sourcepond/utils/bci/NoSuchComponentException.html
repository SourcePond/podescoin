12  	private static String createMessage(final String pFieldName, final int pParameterIndex, final Class<?> pFieldType) {
13  		if (pFieldName != null) {
14  			return format("No component found which matches field '%s' with type '%s'", pFieldName,
15  					pFieldType.getName());
16  		}
17  		return format("No component found which matches parameter at index %d with type '%s'", pParameterIndex,
18  				pFieldType.getName());
19  	}
20  
21  	public NoSuchComponentException(final String pFieldName, final int pParameterIndex, final Class<?> pFieldType) {
22  		super(createMessage(pFieldName, pParameterIndex, pFieldType));
23  	}
