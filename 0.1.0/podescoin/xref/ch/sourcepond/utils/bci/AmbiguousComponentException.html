3   import java.util.Map;
4   
5   public class AmbiguousComponentException extends RuntimeException {
6   
7   	/**
8   	 * 
9   	 */
10  	private static final long serialVersionUID = 1L;
11  
12  	private static String createMessage(final String pFieldNameOrNull, final int pParameterIndex,
13  			final Map<String, Object> pAmbiguousComponents) {
14  		final StringBuilder message = new StringBuilder();
15  		if (pFieldNameOrNull != null) {
16  			message.append("There is more than one component which matches field ").append(pFieldNameOrNull);
17  		} else {
18  			message.append("There is more than one component which matches parameter at index ")
19  					.append(pParameterIndex);
20  		}
21  		message.append(". Following components have been found:\n");
22  		for (final Map.Entry<String, Object> entry : pAmbiguousComponents.entrySet()) {
23  			message.append("\t").append(entry.getKey()).append(" -> ").append(entry.getValue()).append("\n");
24  		}
25  		return message.toString();
26  	}
27  
28  	public AmbiguousComponentException(final String pFieldNameOrNull, final int pParameterIndex,
29  			final Map<String, Object> pAmbiguousComponents) {
30  		super(createMessage(pFieldNameOrNull, pParameterIndex, pAmbiguousComponents));
31  	}
32  }
